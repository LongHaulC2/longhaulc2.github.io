269 // 64 bit integer where 32 bits are used for the hash offset
270 // and remaining 32 bits are used for the hash computed using it
271 using offset_hash_pair = unsigned long long;
272
273 LAZY_IMPORTER_FORCEINLINE constexpr unsigned get_hash(offset_hash_pair pair) noexcept {
274 return (pair & 0xFFFFFFFF);
275 }
276
277 LAZY_IMPORTER_FORCEINLINE constexpr unsigned get_offset(offset_hash_pair pair) noexcept {
278 return static_cast<unsigned>(pair >> 32);
279 }
280
281 template<bool CaseSensitive = LAZY_IMPORTER_CASE_SENSITIVITY>
282 LAZY_IMPORTER_FORCEINLINE constexpr unsigned hash_single(unsigned value, char c) noexcept
283 {
284 return (value ^ static_cast<unsigned>((!CaseSensitive && c >= 'A' && c <= 'Z') ? (c | (1 << 5)) : c)) * 16777619;
285 }
286
287 LAZY_IMPORTER_FORCEINLINE constexpr unsigned
288 khash_impl(const char* str, unsigned value) noexcept
289 {
290 return (*str ? khash_impl(str + 1, hash_single(value, *str)) : value);
291 }
292
293 LAZY_IMPORTER_FORCEINLINE constexpr offset_hash_pair khash(
294 const char* str, unsigned offset) noexcept
295 {
296 return ((offset_hash_pair{ offset } << 32) | khash_impl(str, offset));
297 }
298
299 template<class CharT = char>
300 LAZY_IMPORTER_FORCEINLINE unsigned hash(const CharT* str, unsigned offset) noexcept
301 {
302 unsigned value = offset;
303
304 for (;;) {
305 char c = *str++;
306 if (!c)
307 return value;
308 value = hash_single(value, c);
309 }
310 }
311
312 LAZY_IMPORTER_FORCEINLINE unsigned hash(
313 const win::UNICODE_STRING_T& str, unsigned offset) noexcept
314 {
315 auto first = str.Buffer;
316 const auto last = first + (str.Length / sizeof(wchar_t));
317 auto value = offset;
318 for (; first != last; ++first)
319 value = hash_single(value, static_cast<char>(*first));
320
321 return value;
322 }
323
324 LAZY_IMPORTER_FORCEINLINE forwarded_hashes hash_forwarded(
325 const char* str, unsigned offset) noexcept
326 {
327 forwarded_hashes res{ offset, offset };
328
329 for (; *str != '.'; ++str)
330 res.module_hash = hash_single<true>(res.module_hash, *str);
331
332 ++str;
333
334 for (; *str; ++str)
335 res.function_hash = hash_single(res.function_hash, *str);
336
337 return res;
338 }

395 struct exports_directory {
396 unsigned long _ied_size;
397 const char* _base;
398 const win::IMAGE_EXPORT_DIRECTORY* _ied;
399
400 public:
401 using size_type = unsigned long;
402
403 LAZY_IMPORTER_FORCEINLINE
404 exports_directory(const char* base) noexcept : _base(base)
405 {
406 const auto ied_data_dir = nt_headers(base)->OptionalHeader.DataDirectory[0];
407 _ied = reinterpret_cast<const win::IMAGE_EXPORT_DIRECTORY*>(
408 base + ied_data_dir.VirtualAddress);
409 _ied_size = ied_data_dir.Size;
410 }
411
412 LAZY_IMPORTER_FORCEINLINE explicit operator bool() const noexcept
413 {
414 return reinterpret_cast<const char*>(_ied) != _base;
415 }
416
417 LAZY_IMPORTER_FORCEINLINE size_type size() const noexcept
418 {
419 return _ied->NumberOfNames;
420 }
421
422 LAZY_IMPORTER_FORCEINLINE const char* base() const noexcept {
423 return _base;
424 }
425 LAZY_IMPORTER_FORCEINLINE const win::IMAGE_EXPORT_DIRECTORY* ied() const noexcept
426 {
427 return _ied;
428 }
429
430 LAZY_IMPORTER_FORCEINLINE const char* name(size_type index) const noexcept
431 {
432 return _base + reinterpret_cast<const unsigned long*>(_base + _ied->AddressOfNames)[index];
433 }
434
435 LAZY_IMPORTER_FORCEINLINE const char* address(size_type index) const noexcept
436 {
437 const auto* const rva_table =
438 reinterpret_cast<const unsigned long*>(_base + _ied->AddressOfFunctions);
439
440 const auto* const ord_table = reinterpret_cast<const unsigned short*>(
441 _base + _ied->AddressOfNameOrdinals);
442
443 return _base + rva_table[ord_table[index]];
444 }
445
446 LAZY_IMPORTER_FORCEINLINE bool is_forwarded(
447 const char* export_address) const noexcept
448 {
449 const auto ui_ied = reinterpret_cast<const char*>(_ied);
450 return (export_address > ui_ied && export_address < ui_ied + _ied_size);
451 }

503 // provides the cached functions which use Derive classes methods
504 template<class Derived, class DefaultType = void*>
505 class lazy_base {
506 protected:
507 // This function is needed because every templated function
508 // with different args has its own static buffer
509 LAZY_IMPORTER_FORCEINLINE static void*& _cache() noexcept
510 {
511 static void* value = nullptr;
512 return value;
513 }
514
515 public:
516 template<class T = DefaultType>
517 LAZY_IMPORTER_FORCEINLINE static T safe() noexcept
518 {
519 return Derived::template get<T, safe_module_enumerator>();
520 }
521
522 template<class T = DefaultType, class Enum = unsafe_module_enumerator>
523 LAZY_IMPORTER_FORCEINLINE static T cached() noexcept
524 {
525 auto& cached = _cache();
526 if (!cached)
527 cached = Derived::template get<void*, Enum>();
528
529 return (T)(cached);
530 }
531
532 template<class T = DefaultType>
533 LAZY_IMPORTER_FORCEINLINE static T safe_cached() noexcept
534 {
535 return cached<T, safe_module_enumerator>();
536 }
537 };

588 template<class F = T, class Enum = unsafe_module_enumerator>
589 LAZY_IMPORTER_FORCEINLINE static F get() noexcept
590 {
591 // for backwards compatability.
592 // Before 2.0 it was only possible to resolve forwarded exports when
593 // this macro was enabled
594#ifdef LAZY_IMPORTER_RESOLVE_FORWARDED_EXPORTS
595 return forwarded<F, Enum>();
596#else
597
598 Enum e;
599
600 do {
601#ifdef LAZY_IMPORTER_HARDENED_MODULE_CHECKS
602 if (!e.value->DllBase || !e.value->FullDllName.Length)
603 continue;
604#endif
605
606 const exports_directory exports(e.value->DllBase);
607
608 if (exports) {
609 auto export_index = exports.size();
610 while (export_index--)
611 if (hash(exports.name(export_index), get_offset(OHP)) == get_hash(OHP))
612 return (F)(exports.address(export_index));
613 }
614 } while (e.next());
615 return {};
616#endif
617 }
618
619 template<class F = T, class Enum = unsafe_module_enumerator>
620 LAZY_IMPORTER_FORCEINLINE static F forwarded() noexcept
621 {
622 detail::win::UNICODE_STRING_T name;
623 forwarded_hashes hashes{ 0, get_hash(OHP) };
624
625 Enum e;
626 do {
627 name = e.value->BaseDllName;
628 name.Length -= 8; // get rid of .dll extension
629
630 if (!hashes.module_hash || hash(name, get_offset(OHP)) == hashes.module_hash) {
631 const exports_directory exports(e.value->DllBase);
632
633 if (exports) {
634 auto export_index = exports.size();
635 while (export_index--)
636 if (hash(exports.name(export_index), get_offset(OHP)) == hashes.function_hash) {
637 const auto addr = exports.address(export_index);
638
639 if (exports.is_forwarded(addr)) {
640 hashes = hash_forwarded(
641 reinterpret_cast<const char*>(addr),
642 get_offset(OHP));
643
644 e.reset();
645 break;
646 }
647 return (F)(addr);
648 }
649 }
650 }
651 } while (e.next());
652 return {};
653 }

676 template<class F = T, bool IsSafe = false, class Module>
677 LAZY_IMPORTER_FORCEINLINE static F in(Module m) noexcept
678 {
679 if (IsSafe && !m)
680 return {};
681
682 const exports_directory exports((const char*)(m));
683 if (IsSafe && !exports)
684 return {};
685
686 for (unsigned long i{};; ++i) {
687 if (IsSafe && i == exports.size())
688 break;
689
690 if (hash(exports.name(i), get_offset(OHP)) == get_hash(OHP))
691 return (F)(exports.address(i));
692 }
693 return {};
694 }
